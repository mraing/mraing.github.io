Vue路由重定向

2019年06月12日 92 字 大概 1 分钟 Vue vue

重定向路由，刚进来就默认显示其中的一个子页面。

// 路由属性
redirect: '重定向地址'

// router.js
import Vue from 'vue'
import Router from 'vue-router'
import HelloWorld from '@/components/HelloWorld'
import Switch from '@/components/mint-ui-switch'



Vue.use(Router)

export default new Router({
  routes: [
    {
      path: '/',
      name: 'HelloWorld',
      component: HelloWorld,
      redirect: '/switch',
      children:[{
        path:'/switch',
        component:Switch
      }]
    }
  ]
})
