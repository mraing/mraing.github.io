Vue引入Mint-UI框架

2019年06月12日 52 字 大概 1 分钟 前端 Mint-UI
  • 安装
npm i mint-ui -S
  • 全局引入
// main.js
import Vue from 'vue'
import MintUI from 'mint-ui'
import 'mint-ui/lib/style.css'
import App from './App.vue'

Vue.use(MintUI)

new Vue({
  el: '#app',
  components: { App }
})
